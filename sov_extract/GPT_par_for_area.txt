    """
    –í—ã—Ä–µ–∑–∞–µ—Ç –∑–∞–¥–∞–Ω–Ω—É—é –æ–±–ª–∞—Å—Ç—å (–ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫ –∏–ª–∏ —ç–ª–ª–∏–ø—Å) –∏–∑ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è.

    center_x, center_y = width // 2, height // 2
    size_x, size_y = size

    if shape == 'square':  # –¢–µ–ø–µ—Ä—å –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç –ø—Ä—è–º–æ—É–≥–æ–ª—å–Ω–∏–∫

        upper = max(center_y - size_y // 2, 0)
        right = min(center_x + size_x // 2, width)
        lower = min(center_y + size_y // 2, height)
        return image.crop((left, upper, right, lower))

    elif shape == 'circle':  # –¢–µ–ø–µ—Ä—å –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç —ç–ª–ª–∏–ø—Å

        draw = ImageDraw.Draw(mask)
        draw.ellipse(
            (center_x - size_x // 2, center_y - size_y // 2,

            fill=255
        )
        result = Image.composite(image, Image.new("RGB", image.size, (0, 0, 0)), mask)

        bbox = mask.getbbox()
        return result.crop(bbox)
